section "Definition of a semitopology"

locale semitopology =
  fixes "open" :: "'a set ⇒ bool"
  assumes "⋀ Os . ∀ O ∈ Os . open O ⟹ open (⋃ Os)"
    and "open UNIV"
begin

lemma "open {}"
  ― ‹The axioms imply that the empty set is open›
  by (metis Sup_empty empty_iff semitopology_axioms semitopology_def)

section "Section 2"

lemma l_2_3_2:
  fixes S
  shows "(∀ p ∈ S . ∃ O . open O ∧ p ∈ O ∧ O ⊆ S) ⟷ open S"
proof -
  have "open S" if "∀ p ∈ S . ∃ O . open O ∧ p ∈ O ∧ O ⊆ S"
  proof -
    from ‹∀ p ∈ S . ∃ O . open O ∧ p ∈ O ∧ O ⊆ S›
    obtain f where 1:"∀ p ∈ S . open (f p) ∧ p ∈ f p ∧ f p ⊆ S" 
      ― ‹first we skolemize the existential in the assumption›
      by metis
    from 1 have 2:"S = ⋃ {f p | p . p ∈ S}" by blast
    from 1 have 3:"open (f p)" if "p ∈ S" for p using `p∈S` by auto
    from 2 3 show ?thesis
      by (smt (verit, ccfv_SIG) mem_Collect_eq semitopology_axioms semitopology_def)
  qed
  moreover
  have "∀ p ∈ S . ∃ O . open O ∧ p ∈ O ∧ O ⊆ S" if "open S"
    using that by blast
  ultimately show ?thesis by blast
qed

section "Section 3"

subsection "Subsection 3.1"

definition intersects :: "'a set ⇒ 'a set ⇒ bool" (infix "int" 100)
  where "intersects a b ≡ a ∩ b ≠ {}"

lemma int_commutes:
  fixes S S'
  shows "S int S' = S' int S"
  by (simp add: inf_commute intersects_def)

lemma int_refl:
  fixes S
  shows "S int S = (S ≠ {})"
  by (simp add: local.intersects_def)

lemma int_union:
  fixes X Y Z
  shows "X int (Y ∪ Z) = (X int Y ∨ X int Z)"
  by (simp add: Int_Un_distrib local.intersects_def)

lemma subset_int:
  fixes X X'
  assumes "X ⊆ X'" and "X≠{}"
  shows "X int X'"
  by (metis assms inf.orderE local.intersects_def)

lemma subset_int_other_1:
  fixes X X' Y
  assumes "X ⊆ X'" and "X int Y"
  shows "X' int Y"
  by (metis assms int_commutes int_union subset_Un_eq)

lemma subset_int_other_2:
  fixes X X' Y
  assumes "Y ⊆ Y'" and "X int Y"
  shows "X int Y'"
  by (metis assms int_union subset_Un_eq)

definition transitive
  where "transitive S ≡ ∀ O O' . open O ∧ open O' ∧ S int O ∧ S int O' ⟶ O int O'"

definition topen
  where "topen S ≡ S ≠ {} ∧ open S ∧ transitive S"

subsection "Subsection 3.4"

lemma l_3_4_2_1:
  fixes S S'
  assumes "S' ⊆ S" and "transitive S"
  shows "transitive S'"
  by (metis Int_assoc Int_empty_right assms le_iff_inf local.intersects_def local.transitive_def)

lemma l_3_4_2_2:
  fixes S S'
  assumes "S' ⊆ S" and "topen S" and "S' ≠ {}" and "open S'"
  shows "topen S'"
  using assms l_3_4_2_1 topen_def by force

lemma l_3_4_3_1:
  fixes S S' O O'
  assumes "transitive S" and "transitive S'" and "open S ∨ open S'"
  and "open O" and "open O'" and "O int S" and "S int S'" and "S' int O'"
  shows "O int O'"
  by (metis Int_commute assms local.intersects_def local.transitive_def) 

lemma l_3_4_3_2:
  fixes S S'
  assumes "S int S'" and "transitive S" and "transitive S'" and "open S ∨ open S'"
  shows "transitive (S ∪ S')"
  by (smt (verit) Un_iff assms disjoint_iff_not_equal local.intersects_def local.transitive_def)

subsection "Subsection 3.5"

definition connected where
  "connected Ss ≡ ∀ S ∈ Ss . ∀ S' ∈ Ss . S int S'"

lemma l_3_5_2_1:
  fixes S S'
  assumes "S int S'" and "topen S" and "topen S'"
  shows "topen (S ∪ S')"
  by (metis Un_empty Un_iff Un_upper1 Un_upper2 assms l_2_3_2 l_3_4_3_2 semitopology.topen_def semitopology_axioms)

subsection "Subsection 3.6"

definition intertwined where
  "intertwined p p' == transitive {p, p'}"

lemma def_3_6_1:
  fixes p p'
  shows  "intertwined p p' = (∀ O O' . open O ∧ open O' ∧ p ∈ O ∧ p' ∈ O' ⟶ O int O')"
proof -
  have "intertwined p p'" if "∀ O O' . open O ∧ open O' ∧ p ∈ O ∧ p' ∈ O' ⟶ O int O'"
    by (smt (verit, best) Int_commute inf_right_idem insert_absorb insert_disjoint(2) intersects_def intertwined_def that transitive_def)
  moreover
  have "∀ O O' . open O ∧ open O' ∧ p ∈ O ∧ p' ∈ O' ⟶ O int O'" if "intertwined p p'"
    using intersects_def intertwined_def that transitive_def by auto
  ultimately show ?thesis by auto
qed

lemma l_3_6_4_a:
  fixes S
  shows "transitive S = (∀ p ∈ S . ∀ p' ∈ S . intertwined p p')"
proof
  show "∀ p ∈ S . ∀ p' ∈ S . intertwined p p'" if "transitive S"
    by (simp add: intertwined_def l_3_4_2_1 that)
  show "transitive S" if "∀ p ∈ S . ∀ p' ∈ S . intertwined p p'"
    by (smt (verit, best) that disjoint_iff_not_equal def_3_6_1 semitopology.intersects_def semitopology.transitive_def semitopology_axioms)
qed

lemma l_3_6_4_b:
  fixes S
  shows "transitive S = (∀ p ∈ S . ∀ p' ∈ S . transitive {p, p'})"
  using intertwined_def l_3_6_4_a by auto

definition K where
  ― ‹The community of a point›
  "K p ≡ interior {p' . intertwined p p'}"

lemma interior_open:
  fixes R
  shows "open (interior R)"
proof -
  have "∃ O . open O ∧ O ⊆ interior R ∧ p ∈ O" if "p ∈ interior R" for p

section "Section 5"

subsection "Subsection 5.1"

definition closure where
  "closure R ≡ {p . ∀ O . open O ∧ p ∈ O ⟶ O int R}"

lemma closure_monotone:
  fixes R R'
  assumes "R ⊆ R'"
  shows "closure R ⊆ closure R'"
  using assms closure_def subset_int_other_2 by force

lemma closure_increasing:
  fixes R
  shows "R ⊆ closure R"
  using closure_def local.intersects_def by auto

lemma closure_idempotent:
  fixes R
  shows "closure R = closure (closure R)"
proof
  show "closure R ⊆ closure (closure R)"
    by (simp add: closure_increasing)
next
  show "closure (closure R) ⊆ closure R"
  proof -
    { fix p
      assume "p ∈ closure (closure R)" and "p ∉ closure R"
      hence False
        unfolding closure_def intersects_def by auto }
    thus ?thesis
      by blast
  qed
qed

definition closed where
  "closed R ≡ R = closure R"

definition clopen where
  "clopen R ≡ open R ∧ closed R"

lemma complement_closed_is_open:
  fixes R
  assumes "closed R"
  shows "open (- R)"
proof -
  have "∃ O . open O ∧ O ⊆ - R ∧ p ∈ O" if "p ∈ -R" for p
  proof -
    from `p ∈ - R`
    have "p ∈ - (closure R)"
      using assms closed_def by auto
    hence "∃ O . open O ∧ O ∩ R = {} ∧ p ∈ O"
      using closure_def local.intersects_def by auto
    thus ?thesis
      by (simp add: inf_shunt)
  qed
  thus ?thesis
    by (meson l_2_3_2) 
qed

lemma complement_open_is_closed:
  fixes O
  assumes "open O"
  shows "closed (-O)"
proof -
  have "O ∩ closure (-O) = {}"
    using assms closure_def semitopology.intersects_def semitopology_axioms by fastforce
  thus ?thesis
    by (meson closed_def closure_increasing compl_le_swap1 disjoint_eq_subset_Compl subset_antisym)
qed

lemma empty_closed:
  "closed {}"
  by (metis Compl_UNIV_eq complement_open_is_closed semitopology_axioms semitopology_def)

lemma univ_closed: "closed UNIV"
  using closed_def closure_increasing by blast

lemma intersection_of_closed_is_closed:
  fixes Rs
  assumes "∀ R ∈ Rs . closed R"
  shows "closed (⋂ Rs)"
proof -
  have 1:"⋂Rs = -⋃{-R | R . R ∈ Rs}"
    by (simp add: Setcompr_eq_image)
  thus ?thesis
    by (smt (verit, del_insts) assms complement_closed_is_open complement_open_is_closed mem_Collect_eq semitopology_axioms semitopology_def)
qed

lemma closure_as_intersection:
  fixes R
  shows "closure R = ⋂ {C . closed C ∧ R ⊆ C}"
  by (smt (z3) cInf_eq_minimum closure_idempotent closure_increasing closure_monotone mem_Collect_eq semitopology.closed_def semitopology_axioms) 

lemma interior_of_closed_in_closed:
  fixes C
  assumes "closed C"
  shows "interior C ⊆ C"
    using semitopology.interior_def semitopology_axioms by auto

lemma open_in_interior_of_closure:
  fixes O
  assumes "open O"
  shows "O ⊆ interior (closure O)"
  using assms closure_increasing interior_def by auto

lemma closure_of_interior_of_closed_in_closed:
  fixes C
  assumes "closed C"
  shows "closure (interior C) ⊆ C"
  by (metis assms closed_def closure_monotone interior_of_closed_in_closed)

subsection "Subsection 5.3"

lemma intertwined_iff_in_closure_of_neighbourhood:
  fixes p p'
  shows "intertwined p p' = (∀ O . open O ∧ p ∈ O ⟶ p' ∈ closure O)"
  using closure_def semitopology.int_commutes def_3_6_1 semitopology_axioms by fastforce

lemma intertwined_is_intersection_of_closures_of_neighbourhoods:
  fixes p
  shows "{p' . intertwined p p'} = ⋂ {closure O | O . open O ∧ p ∈ O}"
proof -
  have "⋂ {closure O | O . open O ∧ p ∈ O} = {p' . ∀ O . open O ∧ p∈O ⟶ p' ∈ closure O}"
    by auto
  thus ?thesis
    by (simp add: semitopology.intertwined_iff_in_closure_of_neighbourhood semitopology_axioms)
qed

lemma intertwined_is_intersection_of_closed:
  fixes p
  shows "{p' . intertwined p p'} = ⋂ {C . closed C ∧ p ∈ interior C}"
proof -
  have "⋂ {closure O | O . open O ∧ p ∈ O} = ⋂ {C . closed C ∧ p ∈ interior C}"
  proof -
    have "∃ C . closed C ∧ p ∈ interior C ∧ C ⊆ closure O" if "open O" and "p ∈ O" for O
    proof -
      define C where "C ≡ closure O"
      have "closed C"
        using C_def closed_def closure_idempotent by auto
      moreover have "C ⊆ closure O"
        by (simp add: C_def)
      moreover have "p ∈ interior C"
        using C_def open_in_interior_of_closure that by blast
      ultimately show ?thesis
        by blast
    qed
    hence "⋂ {C . closed C ∧ p ∈ interior C} ⊆ ⋂ {closure O | O . open O ∧ p ∈ O}"
      by fastforce
    moreover
    have "∃ O . open O ∧ p ∈ O ∧ closure O ⊆ C" if "closed C" and "p ∈ interior C" for C
    proof -
      define O where "O ≡ interior C"
      have "open O"
        by (simp add: O_def interior_open)
      moreover have "p ∈ O"
        using O_def that(2) by auto
      moreover have "closure O ⊆ C"
        by (simp add: O_def closure_of_interior_of_closed_in_closed that(1))
      ultimately show ?thesis by blast
    qed
    hence "⋂ {closure O | O . open O ∧ p ∈ O} ⊆ ⋂ {C . closed C ∧ p ∈ interior C}" 
      by fastforce
    ultimately
    show ?thesis by auto
  qed
  thus ?thesis
    by (simp add: semitopology.intertwined_is_intersection_of_closures_of_neighbourhoods semitopology_axioms)
qed

lemma "closed {p' . intertwined p p'}"
  using intersection_of_closed_is_closed intertwined_is_intersection_of_closed by auto

lemma closure_in_intertwined_set:
  "closure {p} ⊆ {p' . intertwined p p'}"
  by (smt (verit) closure_def insert_is_Un int_commutes int_union mem_Collect_eq mk_disjoint_insert
      semitopology.def_3_6_1 semitopology_axioms subsetI)

lemma closure_is_intertwined_set:
  assumes "interior (closure {p}) ≠ {}"
  shows "closure {p} = {p' . intertwined p p'}"
  oops

lemma cascade:
  ― ‹This is a version of the Cascade Theorem›
  fixes S O
  assumes "topen S" and "open O" and "O int S"
  shows "S ⊆ closure O" using closure_increasing
  by (smt (verit, best) assms int_commutes closure_def mem_Collect_eq semitopology.topen_def semitopology_axioms subsetD subsetI transitive_def)

  "topen S ≡ open S ∧ transitive S"

  "strong_topen S ≡ open S ∧ (∀ O O' . open O ∧ open O' ∧ S int O ∧ S int O' ⟶ O ∩ O' ∩ S ≠ {})"

  "intertwined p p' == transitive {p, p'}"

  "intertwined_set_of p ≡ {p' . intertwined p p'}"

  "closed R ≡ R = closure R"

  "interior R ≡ ⋃ {O . open O ∧ O ⊆ R}"

  "boundary R ≡ R - interior R"

  "edge p ≡ boundary (intertwined_set_of p)"

  "K p ≡ interior {p' . intertwined p p'}"

  "regular p ≡ p ∈ K p ∧ topen (K p)"

  "weakly_regular p  ≡  p ∈ (K p)"

  "conflicted p  ≡ ∃ p'. ∃ p''. ((intertwined p' p)  ∧ (intertwined p p'')  ∧ ¬ (intertwined p' p''))"

  "unconflicted p  ≡ ∀ p'. ∀ p''. intertwined p' p  ∧ intertwined p p''   ⟶  intertwined p' p''"

  "dense_in D P  ≡  (D ⊆ P) ∧ (∀ O. open O ∧ O ≠ {} ∧ O ⊆ P  ⟶ D int O)"

  "strongly_dense_in D P  ≡  (D ⊆ P) ∧ (∀ O. open O ∧ P int O  ⟶ D int O)"

  "resilient S ≡ ∀ C . closed C ⟶ strong_topen (S - C)"
